import random

user_score = 0
computer_score = 0


def convert_choice(choice):
    if choice == 'r':
        return "Rock"
    elif choice == 'p':
        return "Paper"
    elif choice == 's':
        return "Scissors"
    else:
        return "Invalid"


def play_round(round_number):
    global user_score, computer_score
    print(f"\nRound No: {round_number}")
    user_input = input("Pick one: r for Rock, p for Paper or s for Scissors: ").lower()
    user_choice = convert_choice(user_input)

    if user_choice == "Invalid":
        print("Wrong entry! Try again.")
        return

    computer_choice = random.choice(["Rock", "Paper", "Scissors"])

    print(f"You chose {user_choice}, Computer chose {computer_choice}.")

    if user_choice == computer_choice:
        print("Result: Tie")
    elif (user_choice == "Rock" and computer_choice == "Scissors") or \
            (user_choice == "Paper" and computer_choice == "Rock") or \
            (user_choice == "Scissors" and computer_choice == "Paper"):
        print("Result: You win!")
        user_score += 1
    else:
        print("Result: You lose!")
        computer_score += 1


def main():
    print("**** Rock Paper Scissors ****")
    print("Rules:")
    print("1. Rock beats Scissors")
    print("2. Scissors beats Paper")
    print("3. Paper beats Rock")

    for round_num in range(1, 4):
        play_round(round_num)

    print("\nGame Over!")
    print(f"Your final score: {user_score}")
    print(f"Computer's final score: {computer_score}")


# Start the game
main()
